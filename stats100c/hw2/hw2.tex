\begin{problem}{}{3}{0}
  We have that $S^2 = \frac{\msum_{i=1}^n (y_i-\bar{y})^2}{n-1}$ and 
  $S_e^2 = \frac{e^{'}e}{n-k-1}=
  \frac{\msum_{i=1}^n(y_i-\bar{y})^2-\msum_{i=1}^n(\hat{y_i}-\bar{y})^2}{n-k-1}$
  \begin{align}
    R_a^2 &= \frac{S^2-s_e^2}{S^2} \\
          &= \frac{n-1}{\msum_{i=1}^n(y_i-\bar{y})^2} 
            \bigg[\frac{\msum_{i=1}^n(y_i-\bar{y})^2}{n-1} - 
            \frac{\msum_{i=1}^n(y_i-\bar{y})^2}{n-k-1} + 
            \frac{\msum_{i=1}^n(\hat{y_i}-\bar{y})^2}{n-k-1} \bigg] \\
          &= \frac{n-1}{\msum_{i=1}^n(y_i-\bar{y})^2} 
            \bigg[\frac{(n-k-1)\msum_{i=1}^n(y_i-\bar{y})^2 -
            (n-1)\msum_{i=1}^n(y_i-\bar{y})^2 + 
            (n-1)\msum_{i=1}^n(\hat{y_i}-\bar{y})^2}{(n-1)(n-k-1)}\bigg] \\

          &= 1 - \frac{(n-1) - (n-1)R^2}{n-k-1} \\
    \Aboxed{R_a^2 &= 1 - \frac{n-1}{n-k-1}(1-R^2)}
  \end{align}
\end{problem}

\begin{problem}{}{7}{0}
  \section{a}
  \textbf{$E[\hat{y}_{n+1}]$:} 
  \begin{align}
    E[\hat{y}_{n+1}] &= E[x_{n+1}^{'}\betah] \\
                     &=x_{n+1}^{'} E[\betah] \\ 
    \Aboxed{E[\hat{y}_{n+1}] &=x_{n+1}^{'}\beta}
  \end{align}
  \textbf{$E[y_{n+1}]$:} 
  \begin{align}
    \Aboxed{E[y_{n+1}] &= E[x_{n+1}^{'}\beta + \varepsilon_{n+1}] = x_{n+1}^{'}\beta} 
  \end{align}
  \section{b}
  \begin{align}
    E(\tilde{y}_{n+1} - y_{n+1}) &= 0 \\
    E[\tilde{y}_{n+1}] -E[y_{n+1}] &= 0 \\
    a^{'}E[Y] - x_{n+1}^{'}\beta &= 0 \\
    a^{'}X \beta - x_{n+1}^{'}\beta &= 0 \\
    \Aboxed{(a^{'}X - x_{n+1}^{'})\beta &= 0 \implies a^{'}X = x_{n+1}^{'}}
  \end{align}
  \section{c}
  \begin{align}
    Var(\hat{y}_{n+1}) &= Var(x_{n+1}^{'}\betah) \\
                       &= x_{n+1}^{'}Var(\betah)x_{n+1} \\
                       &= x_{n+1}^{'}[\sigma^2(X^{'}X)^{-1}]x_{n+1} \\
    \Aboxed{Var(\hat{y}_{n+1}) &= \sigma^2 x_{n+1}^{'}(X^{'}X)^{-1}x_{n+1}}
  \end{align}
  \begin{align}
    \Aboxed{Var(\tilde{y}_{n+1}) &= Var(a^{'}Y) = \sigma^2 a^{'}a}  
  \end{align}
  \section{d}
  \begin{align}
    Var(\tilde{y}_{n+1}) - Var(\hat{y}_{n+1}) &= 0 \\
    \sigma^2 a^{'}a - \sigma^2 x_{n+1}^{'}(X^{'}X)^{-1}x_{n+1} &= \sigma^2 a^{'}a - \sigma^2 a^{'}x(X^{'}X)^{-1}x^{'}a\\
    \sigma^2 a^{'} (I - X(X^{'}X)^{-1}X^{'})a &= \sigma^2 a^{'}(I-H)a  \\
    \sigma^2 a^{'}(I-H)(I-H)^{'}a &= \sigma^2 ((I-H)^{'}a)^{'}(I-H)^{'}a 
  \end{align}
  We notice that $((I-H)^{'}a)^{'}(I-H)^{'}a$ is of the form $(M^{'}l)^{'}(M^{'}l)$ which 
  implies that $(I-H)$ is at least postive semidefinite so then 
  \begin{align}
    \sigma^2 ((I-H)^{'}a)^{'}(I-H)^{'}a \ge 0
  \end{align}
  which implies that
  \begin{align}
    Var(\tilde{y}_{n+1}) - Var(\hat{y}_{n+1}) &\ge 0 \\
    Var(\tilde{y}_{n+1}) &\ge Var(\hat{y}_{n+1})
  \end{align}
\end{problem}

    &= tr[\sigma^2I + X\beta \beta^{'}X^{'} - \sigma^2 H - HX\beta \beta^{'}X^{'}] \\
    &= \sigma^2 tr(I) - \sigma^2 tr(H) \\
    \Aboxed{tr[(I-H)E[Y^{'}Y]] &= \sigma^2 (n-k-1)}
  \end{align}
  \section{c}
  \begin{align}
    Var(e) &= Var((I-H)Y) \\
           &= \sigma^2 (I-H) (I-H)^{'} \\
           &= \sigma^2 (I-H) \\
    Var(e) &= \sigma^2 (I-X(X^{'}X)^{-1}X^{'})
  \end{align}
  Therefore $Var(e_i) = \sigma^2 (1 - x_i^{'}(X^{'}X)^{-1}x_i)$

  \begin{align}
    Var(\hat{Y}) &= Var(H Y) \\
                 &= H Var(Y)H^{'} \\
                 &= H (\sigma^2 I) H^{'} \\
                 &= \sigma^2 H \\
    Var(\hat{Y}) &= \sigma^2 X(X^{'}X)^{-1}X^{'}  
  \end{align}
Therefore $Var(e_i) = \sigma^2 - Var(\hat{Y})$
\end{problem}
